1220. 统计元音字母序列的数目

func countVowelPermutation(n int) (ans int) {
    const mod int = 1e9 + 7
    dp := [5]int{1, 1, 1, 1, 1}
    for i := 1; i < n; i++ {
        dp = [5]int{
            (dp[1] + dp[2] + dp[4]) % mod, // a 前面可以为 e,u,i
            (dp[0] + dp[2]) % mod,         // e 前面可以为 a,i
            (dp[1] + dp[3]) % mod,         // i 前面可以为 e,o
            dp[2],                         // o 前面可以为 i
            (dp[2] + dp[3]) % mod,         // u 前面可以为 i,o
        }
    }
    for _, v := range dp {
        ans = (ans + v) % mod
    }
    return
}

290. 单词规律

func wordPattern(pattern string, s string) bool {
    word2ch := map[string]byte{}
    ch2word := map[byte]string{}
    words := strings.Split(s, " ")
    if len(pattern) != len(words) {
        return false
    }
    for i, word := range words {
        ch := pattern[i]
        if word2ch[word] > 0 && word2ch[word] != ch || ch2word[ch] != "" && ch2word[ch] != word {
            return false
        }
        word2ch[word] = ch
        ch2word[ch] = word
    }
    return true
}
