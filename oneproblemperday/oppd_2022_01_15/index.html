1716. 计算力扣银行的钱

func totalMoney(n int) (ans int) {
    // 所有完整的周存的钱
    weekNum := n / 7
    firstWeekMoney := (1 + 7) * 7 / 2
    lastWeekMoney := firstWeekMoney + 7*(weekNum-1)
    weekMoney := (firstWeekMoney + lastWeekMoney) * weekNum / 2
    // 剩下的不能构成一个完整的周的天数里存的钱
    dayNum := n % 7
    firstDayMoney := 1 + weekNum
    lastDayMoney := firstDayMoney + dayNum - 1
    dayMoney := (firstDayMoney + lastDayMoney) * dayNum / 2
    return weekMoney + dayMoney
}

1345. 跳跃游戏 IV

func minJumps(arr []int) int {
	n := len(arr)
	if n <= 1 {
		return 0
	}
	dic := map[int][]int{} // 哈希表【同值】切片
	for index, i := range arr {
		dic[i] = append(dic[i], index)
	}
	queue := []int{0}                // 广搜队列
	dicLast := map[int]bool{0: true} // 索引记忆，已经去过的下标不再跳跃
	dicNum := map[int]bool{}         // 同值记忆，已经出现过的值不再进行同值跳跃
	cnt := 0
	for len(queue) > 0 {
		cnt++
		lens := len(queue)
		for k := 0; k < lens; k++ {
			i := queue[k]
			if i+1 < n && !dicLast[i+1] {
				dicLast[i+1] = true
				queue = append(queue, i+1)
			}
			if i-1 >= 0 && !dicLast[i-1] {
				dicLast[i-1] = true
				queue = append(queue, i-1)
			}
			for index := len(dic[arr[i]]) - 1; index >= 0 && !dicNum[arr[i]]; index-- {
				// 倒着入队列，更靠近 n-1
				j := dic[arr[i]][index]
				if j != i && !dicLast[j] {
					dicLast[j] = true
					queue = append(queue, j)
				}
			}
			if dicLast[n-1] {
				return cnt
			}
			dicNum[arr[i]] = true
		}
		queue = queue[lens:]
	}
	return 0
}
