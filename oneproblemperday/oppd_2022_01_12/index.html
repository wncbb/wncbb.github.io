334. 递增的三元子序列

func increasingTriplet(nums []int) bool {
    if len(nums)<3{
        return false
    }
    first:=nums[0]
    second:=int(math.MaxInt64)
    for i:=1; i<len(nums); i++{
        cur:=nums[i]
        if cur>second{
            return true
        }else if cur>first{
            second=cur
        }else{
            // 这里只改first是可以的，下一次遍历过程中，如果cur进入"大于first"分支，那么second就会被修改掉
            first=cur
        }
    }
    return false
}

701. 二叉搜索树中的插入操作

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func insertIntoBST(root *TreeNode, val int) *TreeNode {
    if root == nil {
        return &TreeNode{Val: val}
    }
    // 新的数据，不是插入左子树，就是插入右子树
    p := root
    for p != nil {
        if val < p.Val {
            // 数据已经插入进去，直接退出
            if p.Left == nil {
                p.Left = &TreeNode{Val: val}
                break
            }
            // 再去看下一层，应该插入在左子树，还是右子树
            p = p.Left
        } else {
            if p.Right == nil {
                p.Right = &TreeNode{Val: val}
                break
            }
            p = p.Right
        }
    }
    return root
}
