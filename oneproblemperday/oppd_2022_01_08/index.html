89.格雷编码

[最棒题解][https://leetcode-cn.com/problems/gray-code/solution/gray-code-jing-xiang-fan-she-fa-by-jyd/]

 1func grayCode(n int) []int {
 2    // n=0时，就是一个数0
 3    ans:=make([]int, 1, 1<<n)
 4    // a=上一行的数据
 5    // b=上一行的数据翻转，然后每个元素前面加1(比如10， 变成110)
 6    // 这一行的数据=a+b(数组连接)
 7    // curRow=prevRow+Reverse(prevRow).map(v=>v+head)
 8    head:=1
 9    for i:=0; i<n; i++{
10        for j:=len(ans)-1; j>=0; j--{
11            ans=append(ans, head+ans[j])
12        }
13        head=head<<1
14    }
15    return ans
16}
