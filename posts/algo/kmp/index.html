package main

import (
	"fmt"
)

func main() {
	s := "atoddasdfasdtodd"
	p := "abab"
	rst := match(s, p)
	fmt.Printf("rst: %+v\n", rst)
}

func match(s, p string) []int {
	nxt := build(p)
	ret := make([]int, 0)
	// j是p的索引，i是s的索引
	for i, j := 0, 0; i < len(s); i++ {
		for j > 0 && s[i] != p[j] {
			j = nxt[j]
		}
		if s[i] == p[j] {
			j++
		}
		if j == len(p) {
			ret = append(ret, i-len(p)+1)
			j = nxt[j]
		}
	}
	return ret
}

// build would build the next array
// a b a b
// 0 0 0 1 2
// nxt[i]表示，p[:i]字符串，最长的相同的前缀跟后缀的长度
// 比如nxt[3], p[:3]="aba",  前缀a，跟后缀a
// 比如nxt[4], p[:4]="abab", 前缀ab，跟后缀ab
func build(p string) []int {
	ret := make([]int, 0, len(p))
	ret = append(ret, 0)
	ret = append(ret, 0)

	for i, j := 1, 0; i < len(p); i++ {
		for j > 0 && p[i] != p[j] {
			j = ret[j]
		}
		if p[i] == p[j] {
			j++
		}
		// 看的是[0:i]闭区间，也就是i+1位置之前的最长前后缀长度
		ret = append(ret, j)
	}
	fmt.Printf("nxt: %+v\n", ret)
	return ret
}
