package main

type Task struct{
	ID string
	Cost int
	Next []string
	PreCost int
}

func scheduleTask(tasks []*Task)int{
	taskID2Task:=make(map[string]*Task)
	for _, v:=range tasks{
		taskID2Task[v.ID]=v
	}

	myList:=make([]string, 0)
	taskID2Indegree:=make(map[string]int)
	for _, curTask:=range tasks{
    
    if _, ok:=taskID2Indegree[curTask.ID]; !ok{
			taskID2Indegree[curTask.ID]=0
		}
    
    
		for _, curID:=range curTask.Next{
			taskID2Indegree[curID]++
		}
	}

	for curID, curIndegree:=range taskID2Indegree{
		if curIndegree==0{
			myList=append(myList, curID)
		}
	}

	rst:=0

	for len(myList)!=0{
		curTaskID:=myList[0]
		myList=myList[1:]
		curTask:=taskID2Task[curTaskID]
		for _, nextTaskID:=range curTask.Next{
			taskID2Indegree[nextTaskID]--
			if taskID2Indegree[nextTaskID]==0{
				myList=append(myList, nextTaskID)
			}
			nextTask:=taskID2Task[nextTaskID]
			nextTask.PreCost=getMax(nextTask.PreCost, curTask.Cost+curTask.PreCost)
		}
		if len(curTask.Next)==0{
			rst=getMax(rst, curTask.PreCost+curTask.Cost)
		}
	}

	return rst
}

func getMax(a, b int)int{
	if a>b{
		return a
	}
	return b
}
