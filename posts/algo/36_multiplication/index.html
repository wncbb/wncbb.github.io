36_multiplication

2021-03-10 10:23:10

https://leetcode-cn.com/circle/discuss/I8NZUx/

import (
	"fmt"
)

func main() {
	num1 := "11"
	num2 := "12"
	rst := addStrings(num1, num2)
	fmt.Printf("rst: %s\n", rst)
}

func addStrings(num1, num2 string) string {
	lookup := make([]byte, 0, 36)
	for i := byte('0'); i <= '9'; i++ {
		lookup = append(lookup, i)
	}
	for i := byte('a'); i <= 'z'; i++ {
		lookup = append(lookup, i)
	}

	carry := 0

	num1Bytes := []byte(num1)
	num2Bytes := []byte(num2)

	i := len(num1Bytes) - 1
	j := len(num2Bytes) - 1

	reversedRst := make([]byte, 0)

	for i >= 0 || j >= 0 || carry > 0 {
		cur := carry
		if i >= 0 {
			cur += byte2Number(num1Bytes[i])
			i--
		}
		if j >= 0 {
			cur += byte2Number(num2Bytes[j])
			j--
		}
		carry = cur / 36
		cur = cur % 36
		if cur >= 0 && cur < len(lookup) {
			reversedRst = append(reversedRst, lookup[cur])
		}
	}
	return string(reverseBytes(reversedRst))
}

func reverseBytes(s []byte) []byte {
	if len(s) <= 1 {
		return s
	}
	i := 0
	j := len(s) - 1
	for i < j {
		s[i], s[j] = s[j], s[i]
		i++
		j--
	}
	return s
}

func byte2Number(b byte) int {
	if '0' <= b && b <= '9' {
		return int(b - '0')
	}
	if 'a' <= b && b <= 'z' {
		return int(b-'a') + 10
	}
	return 0
}
