 8func f(m [][]int) [][]int {
 9	v := len(m)
10	dist := make([][]int, v, v)
11	for i := 0; i < v; i++ {
12		dist[i] = make([]int, v, v)
13	}
14	for i := 0; i < v; i++ {
15		for j := 0; j < v; j++ {
16			dist[i][j] = m[i][j]
17		}
18	}
19
20	for k := 0; k < v; k++ {
21		for i := 0; i < v; i++ {
22			for j := 0; j < v; j++ {
23				newMinForIJ := dist[i][k] + dist[k][j]
24				if newMinForIJ < dist[i][j] {
25					dist[i][j] = newMinForIJ
26				}
27			}
28		}
29	}
30	return dist
31}
