Merge Sort

 2
 3import(
 4	"fmt"

 7func merge(a, b []int)[]int{
 8	rst:=make([]int, 0, len(a)+len(b))
 9	i:=0
10	j:=0
11	for i<len(a) && j<len(b){
12		if a[i]<b[j]{
13			rst=append(rst, a[i])
14			i++
15		}else{
16			rst=append(rst, b[j])
17			j++
18		}
19	}
20	rst=append(rst, a[i:]...)
21	rst=append(rst, b[j:]...)
22	return rst
23}
24
25func mergeSort(s []int)[]int{
26	if len(s)<=1{
27		return s
28	}
29
30	middleIdx:=len(s)/2
31	left:=mergeSort(s[:middleIdx])
32	right:=mergeSort(s[middleIdx:])
33	return merge(left, right)
34}
