Install tensorflow

https://docs.python.org/3/library/venv.html

python3 -m venv <path>
source <path>/bin/activate

pip install tensorflow==<version>

Use tensorflow/serving

git: https://github.com/tensorflow/serving

doc: https://www.tensorflow.org/tfx/serving/docker

Multi Version(hot reload), Multi Model(cold reload)

docker run -t --rm -p 8501:8501 \
    -v "$TESTDATA:/models"  -d  \
    -t tensorflow/serving --model_config_file=/models/models.config

models.config

model_config_list:{
    config:{
      name:"hello-world",
      base_path:"/models/hello-world",
      model_platform:"tensorflow",
      model_version_policy:{
        all:{}
      }
    },
    config:{
      name:"modelA",
      base_path:"/models/modelA",
      model_platform:"tensorflow",
      model_version_policy:{
        all:{}
      }
    }
}
