/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func buildTree(preorder []int, inorder []int) *TreeNode {
    if len(inorder)==0{
        return nil
    }
   
    rootNode:=&TreeNode{
        Val: preorder[0],
    }
    
    midIdx:=0
    for i:=0; i<len(inorder); i++{
        if inorder[i]==preorder[0]{
            midIdx=i
            break
        }
    }
    // fmt.Printf("inoder:%+v preorder:%+v midIdx:%d\n", inorder, preorder, midIdx)
    // 注意，preorder也需要跟着变化 preorder跟inorder的个数是一样的
    rootNode.Left=buildTree(preorder[1:midIdx+1], inorder[:midIdx])
    rootNode.Right=buildTree(preorder[midIdx+1:], inorder[midIdx+1:])
    
    return rootNode
}
