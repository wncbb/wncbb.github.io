/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseList(head *ListNode) *ListNode {
    var newHead *ListNode
    cur:=head
    var next *ListNode
    for cur!=nil{
        next=cur.Next
        cur.Next=newHead
        newHead=cur
        cur=next
    }
    return newHead
}
