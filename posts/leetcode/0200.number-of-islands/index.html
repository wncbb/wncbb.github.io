func numIslands(grid [][]byte) int {
    rst:=0
    for i:=0; i<len(grid); i++{
        for j:=0; j<len(grid[0]); j++{
            if grid[i][j]=='0'{
                continue
            }
            rst++
            delIsland(grid, i, j)
        }
    }
    return rst
}

func delIsland(grid [][]byte, i, j int){
    // 千万别忘记这一个，如果是0, 直接返回，不然直接死循环
    if grid[i][j]=='0'{
        return
    }
    grid[i][j]='0'
    dirs:=[][]int{{0, 1},{0, -1},{1, 0},{-1, 0}}
    for _, dir:=range dirs{
        newI:=i+dir[0]
        newJ:=j+dir[1]
        if newI>=0 && newI<len(grid) && newJ>=0 && newJ<len(grid[0]){
            delIsland(grid, newI, newJ)
        }
    }
}
