// reverse m~n

package main

import "fmt"

// Definition for singly-linked list.
type ListNode struct {
     Val int
     Next *ListNode
}


func reverseBetween(head *ListNode, m, n int)*ListNode{
	newHead, _:=innerReverseBetween(head, m, n)
	return newHead
}

func innerReverseBetween(head *ListNode, m, n int)(*ListNode, *ListNode){
	dummy:=&ListNode{}
	dummy.Next=head
	prev:=dummy
	for i:=0; i<m-1; i++{
		prev=prev.Next
	}
	cur:=prev.Next
	var nxt *ListNode
	for i:=0; i<n-m; i++{
		nxt=cur.Next
		cur.Next=nxt.Next
		nxt.Next=prev.Next
		prev.Next=nxt
	}

	return dummy.Next, nxt
}

func printList(name string, h *ListNode){
	fmt.Printf("name:%s\n", name)
	for h!=nil{
		fmt.Printf("%d\n", h.Val)
		h=h.Next
	}
}

func main(){
	l1:=&ListNode{Val: 1}
	l2:=&ListNode{Val: 2}
	l3:=&ListNode{Val: 3}
	l4:=&ListNode{Val: 4}
	l5:=&ListNode{Val: 5}

	l1.Next=l2
	l2.Next=l3
	l3.Next=l4
	l4.Next=l5
	printList("start", l1)
	rst:=reverseBetween(l1, 2, 4)
	printList("end", rst)
}
