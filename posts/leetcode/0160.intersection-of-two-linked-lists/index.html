0160.intersection-of-two-linked-lists

2021-03-10 12:26:32 
遍历A+B跟遍历B+A
A=1, 3, 5, 7, 8
B=2, 6, 7, 8

A与B的最后一部分肯定相同，所以第一个相同的就是交叉点
A+B: 1 3 5 7 8 2 6 7 8
B+A: 2 6 7 8 1 3 5 7 8 


/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */

func getIntersectionNode(headA, headB *ListNode) *ListNode {
    pA:=headA
    pB:=headB
    for pA!=pB{
        if pA!=nil{
            pA=pA.Next
        }else{
            pA=headB
        }
        if pB!=nil{
            pB=pB.Next
        }else{
            pB=headA
        }
    }
    return pA
}
