func canJump(nums []int) bool {
    // 记录当前位置之前能够到达的最远索引
    maxPosition:=0
    for i:=0; i<len(nums); i++{
        // 如果i大于当前能够到达的最远索引，说明没法到达了
        // 因为游戏继续进行下去的前提是能够到达i，
        // i>maxPosition 表示当前的跳远最远距离都到达不了索引i了
        if i>maxPosition{
            return false
        }
        maxPosition=getMax(maxPosition, i+nums[i])
    }
    return true
}

func getMax(a, b int)int{
    if a>b{
        return a
    }
    return b
}
