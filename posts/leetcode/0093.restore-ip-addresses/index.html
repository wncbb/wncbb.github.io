0093.restore-ip-addresses

2021-03-10 13:14:56 #leetcode 
func restoreIpAddresses(s string) []string {
	rst := make([]string, 0)

	var dfs func(curRst []string, start int)
	dfs = func(curRst []string, start int) {
		if len(curRst) == 4 {
			if start == len(s) {
				rst = append(rst, strings.Join(curRst, "."))
				return
			} else {
				return
			}
		}

		for length := 1; length <= 3; length++ {
			// 2+2   2 3 4
			if start+length > len(s) {
				return
			}
			curStr := s[start : start+length]
			if length > 1 && curStr[0] == '0' {
				return
			}
			curNum, _ := strconv.ParseInt(curStr, 10, 32)
			if curNum > 255 {
				return
			}
			curRst = append(curRst, curStr)
			dfs(curRst, start+length)
			curRst = curRst[:len(curRst)-1]
		}
	}
	dfs(make([]string, 0, 4), 0)
	return rst
}
