0199.binary-tree-right-side-view

2021-02-26 14:55:03

需要区分二叉树的这一层下一层，准备俩queue，一个当前处理的curQ， 一个q存放从curQ里出来的node的孩子节点 然后curQ=q; q=nil 继续下一次循环

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func rightSideView(root *TreeNode) []int {
    if root==nil{
        return nil
    }
    ret:=make([]int, 0)
    q:=make([]*TreeNode, 0)
    q=append(q, root)
    for len(q)!=0{
        curRst:=0
        curQ:=q 
        q=nil

        for i:=0; i<len(curQ); i++{
            curNode:=curQ[i]
            curRst=curNode.Val
            if curNode.Left!=nil{
                q=append(q, curNode.Left)
            }
            if curNode.Right!=nil{
                q=append(q, curNode.Right)
            }
        }
        ret=append(ret, curRst)
    }

    return ret
}
