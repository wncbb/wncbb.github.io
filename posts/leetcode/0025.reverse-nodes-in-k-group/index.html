0025.reverse-nodes-in-k-group

2021-02-25 15:58:28 
 1func reverseKGroup(head *ListNode, k int) *ListNode {
 2	var innerReverseKGroup func(h *ListNode) *ListNode
 3	innerReverseKGroup = func(h *ListNode) *ListNode {
 4		cur := h
 5		i := 0
 6		var prev *ListNode
 7		for i = 0; i < k && cur != nil; i++ {
 8			prev = cur
 9			cur = cur.Next
10		}
11		if i < k {
12			return h
13		}
14        // prev是上一个链的末尾
15		prev.Next = nil
16		newList := reverseList(h)
17        // cur是当前链的开头
18        // h刚开始是上一个链的头，反转后肯定是上一个链的尾
19		h.Next = innerReverseKGroup(cur)
20
21		return newList
22	}
23	return innerReverseKGroup(head)
24}

26func reverseList(head *ListNode) *ListNode {
27	var newHead *ListNode
28	cur := head
29	for cur != nil {
30		next := cur.Next
31		cur.Next = newHead
32		newHead = cur
33		cur = next
34	}
35	return newHead
36}
