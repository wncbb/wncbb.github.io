0004.median-of-two-sorted-arrays

2020-12-30 20:04:23 
示例 1：

输入：nums1 = [1,3], nums2 = [2]
输出：2.00000
解释：合并数组 = [1,2,3] ，中位数 2
示例 2：

输入：nums1 = [1,2], nums2 = [3,4]
输出：2.50000
解释：合并数组 = [1,2,3,4] ，中位数 (2 + 3) / 2 = 2.5
示例 3：

输入：nums1 = [0,0], nums2 = [0,0]
输出：0.00000
示例 4：

输入：nums1 = [], nums2 = [1]
输出：1.00000
示例 5：

输入：nums1 = [2], nums2 = []
输出：2.00000


 1import (
 2	"math"
 3)
 4
 5func findMedianSortedArrays(nums1 []int, nums2 []int) float64 {
 6	left := (len(nums1) + len(nums2) + 1) / 2
 7	right := (len(nums1) + len(nums2) + 2) / 2
 8	return (float64(findK(nums1, 0, nums2, 0, left)) + float64(findK(nums1, 0, nums2, 0, right))) / 2.0
 9}
10
11// findK 找到两个数组中第k小的数，从1开始计数
12func findK(nums1 []int, i int, nums2 []int, j int, k int) int {
13	if i >= len(nums1) {
14		// 如果nums1越界，那么一定在nums2里
15		return nums2[j+k-1]
16	}
17	if j >= len(nums2) {
18		// 如果nums2越界，那么一定在nums1里
19		return nums1[i+k-1]
20	}
21
22	if k == 1 {
23		// 如果k==1，就是要找第一小的数，谁小返回谁
24		if nums1[i] < nums2[j] {
25			return nums1[i]
26		} else {
27			return nums2[j]
28		}
29	}
30
31	// nums1往前走k/2-1个索引，看看这个数
32	midValA := math.MaxInt64
33	if i+k/2-1 < len(nums1) {
34		midValA = nums1[i+k/2-1]
35	}
36
37	// nums2往前走k/2-1个索引，看看这个数
38	midValB := math.MaxInt64
39	if j+k/2-1 < len(nums2) {
40		midValB = nums2[j+k/2-1]
41	}
42
43	if midValA < midValB {
44		// nums1的往前走k/2-1个索引的数更小，说明nums1[i:i+k/2-1+1]这些数可以抛弃, 因为要找的数一定不在这里
45		return findK(nums1, i+k/2, nums2, j, k-k/2)
46	}
47
48	return findK(nums1, i, nums2, j+k/2, k-k/2)
49}
