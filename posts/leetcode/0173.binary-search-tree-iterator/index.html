0173.binary-search-tree-iterator

2021-03-28 17:18:28

morris中序遍历

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
type BSTIterator struct {
    cur *TreeNode
}


func Constructor(root *TreeNode) BSTIterator {
    return BSTIterator{
        cur: root,
    }
}


func (this *BSTIterator) Next() int {
    var prev *TreeNode
    var ret int
    for this.cur!=nil{
        if this.cur.Left==nil{
            // 中序遍历顺序是左中右，如果左边没有元素了
            // 那么就轮到visit自己，然后cur就到了cur.Right
            ret=this.cur.Val
            this.cur=this.cur.Right
            return ret
        }else{
            // 如果左子树不为空，需要找当前节点的前驱节点，也就是左子树中最大的那个数，在cur的左子树的最右边
            prev=this.cur.Left
            // 因为前驱节点的right可能会指向cur，所以要求prev.Right!=nil， 缺prev.Right!=cur
            for prev.Right!=nil && prev.Right!=this.cur{
                prev=prev.Right
            }
            if prev.Right==nil{
                // 如果右子树为空，说明还没访问到cur的前驱节点，肯定不能访问cur，
                // cur的前驱结点的right指向cur
                prev.Right=this.cur
                // 然后cur=cur.Left
                this.cur=this.cur.Left
            }else{
                // 如果右子树等于cur，说明prev已经访问过了，所以需要visit cur
                ret=this.cur.Val
                // 恢复前驱结点的right为nil
                prev.Right=nil
                // 访问完cur，下一个是cur的right
                this.cur=this.cur.Right
                return ret
            }
        }
    }
    return -1
}


func (this *BSTIterator) HasNext() bool {
    return this.cur!=nil
}

 * obj := Constructor(root);
 * param_1 := obj.Next();
 * param_2 := obj.HasNext();
 */
