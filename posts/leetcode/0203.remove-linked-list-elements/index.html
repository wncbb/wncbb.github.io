0203.remove-linked-list-elements

2021-03-12 21:31:49 
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func removeElements(head *ListNode, val int) *ListNode {
    dummy:=&ListNode{}
    dummy.Next=head
    // tail:=dummy
    cur:=dummy
    for cur!=nil{
        for cur.Next!=nil && cur.Next.Val==val{
            cur.Next=cur.Next.Next
        }
        cur=cur.Next
    }
    return dummy.Next
}
