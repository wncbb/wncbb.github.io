0030.substring-with-concatenation-of-all-words

2020-12-18 19:56:59 
有点暴力的解法，就是遍历所有可能的窗口

 1func findSubstring(s string, words []string) []int {
 2	//声明返回值
 3	var result []int
 4
 5	//判断
 6	if len(s) == 0 || len(words) == 0 {
 7		return result
 8	}
 9
10
11	//获取单词长度
12	l := len(words[0])
13
14	//获取滑动窗口总长度
15	length := l * len(words)
16
17	//判断
18	if length > len(s) {
19		return result
20	}
21
22	//构造map
23	mp := make(map[string]int)
24
25	//统计
26	for _,v := range words {
27		mp[v] += 1
28	}
29
30	//开始遍历
31	for i := 0; i <= (len(s) - length);i++ {
32		//判断i开始length位置
33		tmpMap := map[string]int{}
34
35		//定义标志
36		//默认是true
37		fg := true
38
39		// 这里的逻辑是，如果当前的单词是需要的，且个数还没凑够，那就对应的数减一
40		// 如果遇到不需要的单词，一定就不合法了.因为需要完美匹配
41		for j := i; j < i + length; j += l {
42			str := string(s[j: j + l])
43			if mp[str] <= 0 {
44                // 这里还没减少，也就是说，必须当前的单词str是需要的，且mp里个数还充足
45				fg = false
46				break
47			} else {
48				mp[str] --
49				tmpMap[str] ++
50			}
51
52		}
53
54		//判断标志
55		if fg == true{
56			result = append(result,i)
57		}
58		
59		//恢复mp
60		for k,v := range tmpMap {
61			mp[k] += v
62		}
63
64	}
65	//返回
66	return result
67
68}
