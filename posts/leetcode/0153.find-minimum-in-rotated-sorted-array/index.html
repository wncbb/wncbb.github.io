0153.find-minimum-in-rotated-sorted-array

2021-03-11 23:18:37 
func findMin(nums []int) int {
    left:=0
    right:=len(nums)-1

    if nums[left]<nums[right]{
        return nums[0]
    }

    for left<right{
        mid:=left+(right-left)/2
        if nums[mid]>nums[right]{
            left=mid+1
        }else{
            right=mid
        }
    }

    // 这种写法不对，因为这个数组是升序排列的，所以得用nums[mid]>nums[right]来做判断
    // for left<right{
    //     fmt.Printf("BEFORE left:%d right:%d\n", left, right)
    //     mid:=left+(right-left)/2
    //     if nums[mid]>nums[left]{
    //         left=mid+1
    //     }else{
    //         right=mid
    //     }
    //     fmt.Printf("AFTER left:%d right:%d\n", left, right)

    // }
    return nums[left]
}
