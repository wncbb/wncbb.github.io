1206.design-skiplist

2020-12-30 16:37:20 
  1import (
  2	"math/rand"
  3)
  4
  5const (
  6	maxLevel     = 16
  7	skipListPVal = 0.5
  8)
  9
 10// randomLevel 返回一个Next数组的高度, 高度不能超过最大值
 11func randomLevel() int {
 12	retLevel := 1
 13	for retLevel < maxLevel && rand.Float32() < skipListPVal {
 14		retLevel++
 15	}
 16	return retLevel
 17}
 18
 19type Node struct {
 20	Val  int
 21	Next [maxLevel]*Node
 22}
 23
 24func NewNode(val int, next [maxLevel]*Node) *Node {
 25	return &Node{
 26		Val:  val,
 27		Next: next,
 28	}
 29}
 30
 31type Skiplist struct {
 32	head     *Node
 33	maxLevel int
 34}
 35
 36func Constructor() Skiplist {
 37	return Skiplist{
 38		// head 里的值是最小值,这里取-1
 39		head:     NewNode(-1, [maxLevel]*Node{}),
 40		maxLevel: 1,
 41	}
 42}
 43
 44func (this *Skiplist) search(target int) [maxLevel]*Node {
 45	cur := this.head
 46	// prevs存，target之前的各层节点
 47	prevs := [maxLevel]*Node{}
 48	maxLevel := this.maxLevel
 49	for i := maxLevel - 1; i >= 0; i-- {
 50		for cur.Next[i] != nil && cur.Next[i].Val < target {
 51			cur = cur.Next[i]
 52		}
 53		// cur.Next[i] 第一个不小于目标值的节点,或者说第一个大于等于目标值的节点
 54		// cur 最大的小于目标值的节点
 55		// 新插入的节点，要在prevs之后
 56		// prevs[i]存的是，第i层,target之前的节点
 57		prevs[i] = cur
 58	}
 59
 60	return prevs
 61}
 62
 63func (this *Skiplist) Search(target int) bool {
 64	prevs := this.search(target)
 65	// 第0层的prevs的节点的下一个节点
 66	return prevs[0].Next[0] != nil && prevs[0].Next[0].Val == target
 67}
 68
 69func (this *Skiplist) Add(num int) {
 70	prevs := this.search(num)
 71	level := randomLevel()
 72	if level > this.maxLevel {
 73		// 如果当前节点的level数目比整体list的都高，需要把prevs高位置的指针换成head
 74		for i := this.maxLevel; i < level; i++ {
 75			prevs[i] = this.head
 76		}
 77		this.maxLevel = level
 78	}
 79	cur := NewNode(num, prevs)
 80
 81	// from prev->next to prev->cur->next
 82	for i := level - 1; i >= 0; i-- {
 83		cur.Next[i] = prevs[i].Next[i]
 84		prevs[i].Next[i] = cur
 85	}
 86}
 87
 88func (this *Skiplist) Erase(num int) bool {
 89	prevs := this.search(num)
 90	// 最底下那一层，就是所有数据都链在一起的，数据不会缺失。这种都搜不都，就一定搜不到
 91	if prevs[0].Next[0] == nil || prevs[0].Next[0].Val != num {
 92		// 如果数据不存在，就不用删
 93		return false
 94	}
 95
 96	del := prevs[0].Next[0]
 97	// from prev->cur->next to prev->next
 98	for i := 0; i < this.maxLevel; i++ {
 99		if prevs[i].Next[i] == del {
100			prevs[i].Next[i] = del.Next[i]
101		}
102	}
103
104	// 每删除一个元素，需要看看list的level层数有没有变化
105	// 就是看看最高层，head的下一个是不是nil
106	for this.maxLevel > 1 && this.head.Next[this.maxLevel-1] == nil {
107		this.maxLevel--
108	}
109
110	return true
111}
