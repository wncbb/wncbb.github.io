0014.longest-common-prefix

2020-12-17 17:22:10 
func longestCommonPrefix(strs []string) string {
    if len(strs)==0{
        return ""
    }
    if len(strs)==1{
        return strs[0]
    }
    

    for i:=0; i<len(strs[0]); i++{
        for j:=1; j<len(strs); j++{
            // fmt.Printf("i:%d strs[j]:%s\n", i, strs[j])
            // 等于号，因为要取strs[j][i], 要是长度是i，最多取到i-1
            if len(strs[j])<=i || strs[j][i]!=strs[0][i]{
                return strs[0][:i]
            } 
        }
    }
    return strs[0]
    
}
