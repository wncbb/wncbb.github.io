0016.3sum-closest

2021-03-12 19:53:23 
import(
    "math"
)

func threeSumClosest(nums []int, target int) int {
    sort.Ints(nums)
    best:=math.MaxInt32
    n:=len(nums)

    for i:=0; i<n-2;i++{
        if i>0 && nums[i]==nums[i-1]{
            continue
        }
        left:=i+1
        right:=n-1
        
        for left<right{
            curSum:=nums[left]+nums[right]+nums[i]
            if curSum==target{
                return target
            }
            if curSum<target{
                left++
            }else{
                right--
            }
            if abs(best-target)>abs(curSum-target){
                best=curSum
            }
        }
    }
    return best
}

func abs(x int) int {
    if x < 0 {
        return -1 * x
    }
    return x
}
