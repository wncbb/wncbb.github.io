0336.palindrome-pairs

2020-12-18 19:36:38 
 1func palindromePairs(words []string) [][]int {
 2	m := make(map[string]int, len(words))
 3	for idx, v := range words {
 4		m[v] = idx
 5	}
 6
 7	res := make([][]int, 0)
 8
 9	for idx, v := range words {
10		// 如果非空字符串v是回文串，且字符数组中包含空字符串，那么  v+"" 与 ""+v 都是回文串
11		if emptyStrIdx, ok := m[""]; ok && isPalindrome(v) && v != "" {
12			res = append(res, []int{emptyStrIdx, idx})
13			res = append(res, []int{idx, emptyStrIdx})
14		}
15		// v的逆序字符串存在的话，那么这俩一拼也是回文串，注意这里只加一次(
16		//  比如A的逆序是B, 遍历到A的话，加入A,B;
17		//  遍历到B的时候，加入B,A，就不会重复
18		// )
19        // 其实v != getReverse(v) 也可以换成 idx!=reverseIdx, 自己如果是回文串，那么一定找不到不是自己的逆序，
20        // 因为回文串的逆序就是它自己
21		if reverseIdx, ok := m[getReverse(v)]; ok && v != getReverse(v) {
22			res = append(res, []int{idx, reverseIdx})
23		}
24		for i := 1; i < len(v); i++ {
25			// 这里直接肢解v，且肢解的两部分不能包含空字符串
26			s1, s2 := divideStr(v, i)
27			// 想要拼一个  s1 s2 x, 那么要求就是s2是回文串，x是s1的逆序，且自己不能跟自己连接
28			if anotherIdx, ok := m[getReverse(s1)]; ok && isPalindrome(s2) && idx != anotherIdx {
29				tmp := []int{idx, anotherIdx}
30				res = append(res, tmp)
31			}
32			// 想要拼一个  x s1 s2，那么要求s1是回文串，且x是s2的逆序，且自己不能跟自己连接
33			if anotherIdx, ok := m[getReverse(s2)]; ok && isPalindrome(s1) && idx != anotherIdx {
34				tmp := []int{anotherIdx, idx}
35				res = append(res, tmp)
36			}
37		}
38	}
39	return res
40}

42func isPalindrome(s string) bool {
43	if s == "" {
44		return true
45	}
46	if len(s) == 1 {
47		return true
48	}
49	sRune := []rune(s)
50	i := 0
51	j := len(sRune) - 1
52	for i < j {
53		if sRune[i] != sRune[j] {
54			return false
55		}
56		i++
57		j--
58	}
59	return true
60}

62func getReverse(input string) string {
63	if len(input) < 2 {
64		return input
65	}
66	n := 0
67	rune := make([]rune, len(input))
68	for _, r := range input {
69		rune[n] = r
70		n++
71	}
72	rune = rune[0:n]
73	// Reverse
74	for i := 0; i < n/2; i++ {
75		rune[i], rune[n-1-i] = rune[n-1-i], rune[i]
76	}
77	// Convert back to UTF-8.
78	output := string(rune)
79	return output
80}

82func divideStr(s string, c int) (string, string) {
83	return string(s[:c]), string(s[c:])
84}
