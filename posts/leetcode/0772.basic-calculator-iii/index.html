 11// 应对-1+(-2+3)这种情况，会转化为 0-1+(0-2+3)
 12func normalS(s string)string{
 13    s=strings.ReplaceAll(s, " ", "")
 14    if s[0]=='-'{
 15        s="0"+s
 16    }
 17    sBytes:=[]byte(s)
 18    rst:=make([]byte, 0)
 19    for i:=0; i<len(s); i++{
 20        if i>0 && (sBytes[i]=='-' || sBytes[i]=='+') && sBytes[i-1]=='('{
 21            rst=append(rst, '0')
 22        }
 23        rst=append(rst, sBytes[i])
 24    }
 25    return string(rst)
 26}

 28func calRPN(s []interface{})int{
 29    myStack:=make([]int, 0)
 30    i:=0
 31    for i<len(s){
 32        if num, ok:=s[i].(int); ok{
 33            myStack=append(myStack, num)
 34        }else if b, ok:=s[i].(byte); ok{
 35            num2:=myStack[len(myStack)-1]
 36            num1:=myStack[len(myStack)-2]
 37            myStack=myStack[:len(myStack)-2]
 38            switch b{
 39            case '+':
 40                myStack=append(myStack, (num1+num2))
 41            case '-':
 42                myStack=append(myStack, (num1-num2))
 43            case '*':
 44                myStack=append(myStack, (num1*num2))
 45            case '/':
 46                myStack=append(myStack, (num1/num2))
 47            }
 48        }
 49        i++
 50    }
 51    return myStack[0]
 52}

 54func genRPN(s string)[]interface{}{
 55
 56    sBytes:=[]byte(s)
 57    rst:=make([]interface{}, 0)
 58    myStack:=&Stack{
 59        raw: make([]byte, 0),
 60    }
 61        
 62    i:=0
 63    for i<len(sBytes){
 64        if sBytes[i]==' '{
 65            i++
 66        }else if isNumber(sBytes[i]){
 67            // 如果是数字，直接输出到结果
 68            curNum:=0
 69            j:=0
 70            for j=i; j<len(sBytes) && isNumber(sBytes[j]); j++{
 71                // fmt.Printf("getNumber(sBytes[%d]): %d\n", j, getNumber(sBytes[j]))
 72                curNum=curNum*10+getNumber(sBytes[j])
 73            }
 74            rst=append(rst, curNum)
 75            i=j
 76        }else if sBytes[i]=='('{
 77            // 如果是左括号，入栈
 78            myStack.Push(sBytes[i])
 79            i++
 80        }else if sBytes[i]==')'{
 81            // 如果是右括号，持续出栈到结果，直到遇到右括号
 82            for myStack.Top()!='('{
 83                rst=append(rst, myStack.Pop()) 
 84            }
 85            myStack.Pop()
 86            i++
 87        }else{
 88            // 如果是计算符号，栈顶计算符优先级如果大于等于当前运算符，需要出栈到结果
 89            // 很好理解，运算符>=当前运算符的，需要先计算，因此需要先出栈到结果中
 90            for !myStack.Empty() && getPriority(myStack.Top())>=getPriority(sBytes[i]){
 91                rst=append(rst, myStack.Pop())
 92            }
 93            myStack.Push(sBytes[i])
 94            i++
 95        }
 96        fmt.Printf("myStack: %+v\n", myStack)
 97    }
 98
 99    for !myStack.Empty(){
100        rst=append(rst, myStack.Pop())
101    }
102
103    return rst
104}

106func getPriority(v byte)int{
107    switch v{
108        case '+':
109            return 1
110        case '-':
111            return 1
112        case '*':
113            return 2
114        case '/':
115            return 2
116    }
117    return 0
118}

120type Stack struct{
121    raw []byte
122}
123func(s *Stack)Push(v byte){
124    s.raw=append(s.raw, v)
125}
126func(s *Stack)Pop()byte{
127    ret:=s.raw[len(s.raw)-1]
128    s.raw=s.raw[:len(s.raw)-1]
129    return ret
130}
131func(s *Stack)Top()byte{
132    return s.raw[len(s.raw)-1]
133}
134func(s *Stack)Empty()bool{
135    return len(s.raw)==0
136}
