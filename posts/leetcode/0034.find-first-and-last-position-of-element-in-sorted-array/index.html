2021-03-12 16:47:38 
func searchRange(nums []int, target int) []int {
    left:=findLeft(nums, target)
    if left==-1{
        return []int{-1, -1}
    }
    right:=findRight(nums, target)
    return []int{left, right}
}

func findLeft(nums []int, target int) int{
    left:=0
    right:=len(nums)-1
    for left<=right{
        fmt.Printf("LEFT left:%d right:%d\n", left, right)
        mid:=left+(right-left)/2
        if nums[mid]==target{
            right=mid-1
        }else if nums[mid]<target{
            left=mid+1
        }else{
            right=mid-1
        }
    }
    if left<len(nums) && nums[left]==target{
        return left
    }
    return -1
}

func findRight(nums []int, target int)int{
    left:=0
    right:=len(nums)-1
    for left<=right{
        fmt.Printf("RIGHT left:%d right:%d\n", left, right)
        mid:=left+(right-left)/2
        if nums[mid]==target{
            left=mid+1
        }else if nums[mid]<target{
            left=mid+1
        }else{
            right=mid-1
        }
    }
    if right>=0 && nums[right]==target{
        return right
    }
    return -1
}
