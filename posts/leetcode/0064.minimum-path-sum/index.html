0064.minimum-path-sum

2021-03-10 21:17:31 
func minPathSum(grid [][]int) int {
    m:=len(grid)
    n:=len(grid[0])
    dp:=make([][]int, 0, m)
    for i:=0; i<m; i++{
        dp=append(dp, make([]int, n, n))
    }
    dp[0][0]=grid[0][0]
    for i:=1; i<m; i++{
        dp[i][0]+=dp[i-1][0]+grid[i][0]
    }
    for j:=1; j<n; j++{
        dp[0][j]+=dp[0][j-1]+grid[0][j]
    }

    // printM(dp)

    for i:=1; i<m; i++{
        for j:=1; j<n; j++{
            dp[i][j]=getMin(dp[i-1][j], dp[i][j-1])+grid[i][j]
        }
    }

    return dp[m-1][n-1]
}

func getMin(a int, s ... int)int{
    ret:=a 
    for _, v:=range s{
        if ret>v{
            ret=v
        }
    }
    return ret

}
