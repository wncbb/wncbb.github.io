1. Max Collatz

2021-09-10 13:35:21

References

  1. https://pe.metaquant.org/pe014.html

Problem

对于所有正整数，考虑如下迭代序列：

if n%2==0{
    n=n/2
}else{
    n=n*3+1
}

经过有限次步骤，一定能到1

现在给定一个数n，求[1, n]中，步骤最大的数

Key

  1. 用缓存存储中间结果，方便后面数字使用
  2. 对于3*n+1，一定是偶数，可以一次算两步

Codes

Golang

package main

import "fmt"

func maxCollatz(n int64) (int64, int64) {
	if n == 1 {
		return 1, 0
	}
	num2Step := make(map[int64]int64)
	num2Step[1] = 0
	rstStep := int64(0)
	rstNum := int64(0)
	for i := int64(1); i <= n; i++ {
		curNum := i
		curStep := int64(0)
		for curNum != 1 {
			if curNum < i {
				curStep += num2Step[curNum]
				break
			} else if curNum%2 == 0 {
				curNum = curNum / 2
				curStep += 1
			} else {
				curNum = (curNum*3 + 1) / 2
				curStep += 2
			}
		}
		fmt.Printf("i:%d curStep:%d curNum:%d\n", i, curStep, curNum)
		num2Step[i] = curStep
		if rstStep < curStep {
			rstStep = curStep
			rstNum = curNum
		}
	}
	return rstNum, rstStep
}

func main() {
	n := int64(3)
	rstNum, rstStep := maxCollatz(n)
	fmt.Printf("rst(%d): rstNum:%d rstStep:%d\n", n, rstNum, rstStep)
}
